Table User {
  user_id PK [uuid]
  email [varchar(255), unique, not null]
  password_hash [varchar(255), not null]
  created_at [timestamp, default: `now()`]
  updated_at [timestamp, default: `now()`]
  deleted_at [timestamp, null]  // ✅ 软删除标记
}

Table UserInfo {
  user_id PK [uuid, ref: > User.user_id, delete: cascade]
  username [varchar(50), not null]
  synopsis [text, null]
  avatar_url [varchar(255), null]
  created_at [timestamp, default: `now()`]
  updated_at [timestamp, default: `now()`]
}

Table Note {
  note_id PK [uuid]
  owner_id [uuid, not null, ref: > User.user_id, delete: restrict]  // ⚠️ 禁止删除有笔记的用户
  title [varchar(200), not null]
  content [text, null]
  version [int, default: 1, not null]
  created_at [timestamp, default: `now()`]
  updated_at [timestamp, default: `now()`]
  deleted_at [timestamp, null]  // ✅ 软删除标记
}

Table Note_User_Permission {
  note_id [uuid, not null, ref: > Note.note_id, delete: cascade]  // ✅ 笔记删除，权限删除
  user_id [uuid, not null, ref: > User.user_id, delete: cascade]  // ✅ 用户删除，权限删除
  permission_level [varchar(20), not null]
  granted_by [uuid, not null, ref: > User.user_id, delete: set null]  // ⚠️ 授权人删除，保留记录
  created_at [timestamp, default: `now()`]
  updated_at [timestamp, default: `now()`]
  
  PK (note_id, user_id)
}

Table Note_Team_Permission {
  note_id [uuid, not null, ref: > Note.note_id, delete: cascade]  // ✅ 笔记删除，权限删除
  team_id [uuid, not null, ref: > Team.team_id, delete: cascade]  // ✅ 团队删除，权限删除
  permission_level [varchar(20), not null]
  granted_by [uuid, not null, ref: > User.user_id, delete: set null]  // ⚠️ 授权人删除，保留记录
  created_at [timestamp, default: `now()`]
  updated_at [timestamp, default: `now()`]
  
  PK (note_id, team_id)
}

Table Team {
  team_id PK [uuid]
  name [varchar(100), not null]
  description [text, null]
  owner_id [uuid, not null, ref: > User.user_id, delete: restrict]  // ⚠️ 禁止删除有团队的用户
  created_at [timestamp, default: `now()`]
  updated_at [timestamp, default: `now()`]
  deleted_at [timestamp, null]  // ✅ 软删除标记
}

Table Team_members {
  team_id [uuid, not null, ref: > Team.team_id, delete: cascade]  // ✅ 团队删除，成员关系删除
  user_id [uuid, not null, ref: > User.user_id, delete: cascade]  // ✅ 用户删除，成员关系删除
  joined_at [timestamp, default: `now()`]
  
  PK (team_id, user_id)
}
