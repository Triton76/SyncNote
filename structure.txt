Table Note {
	note_Id Primary,
	owner_Id not null,
	title,
	content,
	version,
	created_at,
	updated_at
}


Table Note_User_Permission {
	note_Id ref Note(note_Id),
	user_Id ref User(user_Id),
	permission_Level not null,
	granted_by ref User(user_Id),
	created_at default now(),
	updated_at default now(),
	PRIMARY KEY (note_Id, user_Id)
}
Table Note_Team_Permission {
	note_Id ref Note(note_Id),
	team_Id ref Team(team_Id),
	permission_Level not null,
	granted_by ref User(user_Id),
	created_at default now(),
	updated_at default now(),
	PRIMARY KEY (note_Id, team_Id)
}

Table User {
	user_Id primary,
	email Not Null Unique,
	password Not null
	created_at,
	updated_at,
}

Table UserInfo{
	user_Id primary ref User(user_Id),
	synopsis,
	avatar,
	username not null
	created_at,
	updated_at,
}

Table Team {
	team_Id primary,
	name not null,
	desc,
	owner_Id not null ref User(user_Id),
	created_at,
	updated_at,
}
Table Team_members {
	team_Id ref Team(team_Id),
	user_Id ref User(user_Id),
	PRIMARY KEY (team_Id, user_Id)
}
